Correlación

Simulación de variables aleatorias X e Y

library(mvtnorm) ## Librería que simula datos de una distribución normal
positiva =  matrix(c(1,0.85,0.85,1), ncol = 2)    # Matrix de correlación, con correlación r=0.85 positiva
negativa =  matrix(c(1,-0.85,-0.85,1), ncol = 2)  # Matrix de correlación, con correlación negativa r = -0.85
sinCorrelacion = matrix(c(1,0,0,1), ncol = 2)     # Matriz de correlación, sin correlación  r = 0

set.seed(1) ## Semilla para generar numeros aleatorios

### Simulamos los datos
## Simulando comportamiento no lineal
x = seq(0, pi, length.out = 100)
y = vector()
for (i in 1:100){
y[i] = sin(x[i]) + rnorm(mean = sin(x[i]), sd =  0.2, n = 1)
}

p1 = rmvnorm(n = 100, mean = c(0,0), sigma = positiva)  ## Simulando 2 variables aleatorias con correlacion positiva
p2 = rmvnorm(n = 100, mean = c(0,0), sigma = negativa)  ## Simulando 2 variables aleatorias con correlacion negativa
p3 = rmvnorm(n = 100, mean = c(0,0), sigma = sinCorrelacion) ## Simulando 2 variables aleatorias sin correlación

Generamos lo gráficos

par(mfrow = c(2,2))  ## Matriz de gráficos 2x2
plot(p1[,1], p1[,2], pch = 19, main = "Correlación positiva", xlab = "variable X", ylab = "Variable Y")
fit1 = lm(p1[,2] ~ p1[,1])
lines(p1[,1], predict(fit1), col = 2, lwd = 2)
text(-1,2, paste("Cor. Pearson=",round(cor(p1[,1],p1[,2]),3)), cex = 0.6)
plot(p2[,1], p2[,2], pch = 19,  main = "Correlación negativa", xlab = "variable X", ylab = "Variable Y")
text(-1,-2, paste("Cor. Pearson=",round(cor(p2[,1],p2[,2]),3)), cex = 0.6)
fit2 = lm(p2[,2] ~ p2[,1])
lines(p2[,1], predict(fit2), col = 2, lwd = 2)
plot(p3[,1], p3[,2], pch = 19,  main = "Sin correlación", xlab = "variable X", ylab = "Variable Y")
text(-2,-2, paste("Cor. Pearson=",round(cor(p3[,1],p3[,2]),3)), cex = 0.6)
fit3 = lm(p3[,2] ~ p3[,1])
lines(p3[,1], predict(fit3), col = 2, lwd = 2)
plot(x, y, main = "Relación No lineal", pch = 19)
fit = lm(y~x+ I(x^2))
lines(x, predict(fit), col = 2, lwd = 2)
text(1, 0, paste("Cor. Pearson=",  round(cor(x,y),3) ), cex = 0.6)
